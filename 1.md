// ═══════════════════════════════════════
// 📥 تحميل الملفات
// ═══════════════════════════════════════
async function downloadFile(url, dest) {
  const res = await fetch(url);
  if (!res.ok) throw new Error(`Download failed: ${url}`);
  const fileStream = fs.createWriteStream(dest);
  await finished(Readable.fromWeb(res.body).pipe(fileStream));
}

// ═══════════════════════════════════════
// 🔤 تحميل وتخزين الخطوط
// ═══════════════════════════════════════
async function ensureFont(fontFamily, fontWeight) {
  // اختيار ملف الخط بناءً على السُمك
  let fontKey = fontFamily || "Amiri";
  if (fontKey === "Amiri" && fontWeight >= 600) fontKey = "Amiri-Bold";
  
  const safeName = fontKey.replace(/[^a-zA-Z0-9-]/g, "_");
  const fontPath = path.resolve(FONTS_DIR, `${safeName}.ttf`);
  
  if (fs.existsSync(fontPath)) return fontPath;
  
  const url = FONT_MAP[fontKey] || FONT_MAP["Amiri"];
  try {
    await downloadFile(url, fontPath);
    return fontPath;
  } catch (e) {
    console.warn(`⚠️ Font download failed for ${fontKey}, falling back to Amiri`);
    const fallback = path.resolve(FONTS_DIR, "Amiri.ttf");
    if (!fs.existsSync(fallback)) {
      await downloadFile(FONT_MAP["Amiri"], fallback);
    }
    return fallback;
  }
}

// ═══════════════════════════════════════
// 🔊 مدة الصوت
// ═══════════════════════════════════════
async function getAudioDuration(filePath) {
  try {
    const { stdout } = await execAsync(`ffprobe -v error -show_entries format=duration -of csv=p=0 "${filePath}"`);
    return parseFloat(stdout.trim()) || 5;
  } catch { return 5; }
}

// ═══════════════════════════════════════
// 📝 تقسيم النص العربي بشكل ذكي
// ═══════════════════════════════════════
function wrapArabicText(text, fontSize, maxWidth) {
  if (!text) return [];
  
  // تقدير عرض الحرف العربي بناءً على حجم الخط
  // الحروف العربية مع التشكيل (حركات) أنحف بكثير من اللاتينية
  // ~0.30 من حجم الخط هو التقدير الأدق لخط أميري مع تشكيل
  const charWidth = fontSize * 0.30;
  const maxCharsPerLine = Math.max(15, Math.floor(maxWidth / charWidth));
  
  const words = text.split(" ");
  const lines = [];
  let currentLine = "";
  
  for (const word of words) {
    const testLine = currentLine ? `${currentLine} ${word}` : word;
    if (testLine.length > maxCharsPerLine && currentLine) {
      lines.push(currentLine.trim());
      currentLine = word;
    } else {
      currentLine = testLine;
    }
  }
  if (currentLine.trim()) lines.push(currentLine.trim());
  return lines;
}

// ═══════════════════════════════════════
// 🎨 توليد SVG layer لكل آية — مطابق للموقع
// ═══════════════════════════════════════
function escapeXml(str) {
  if (!str) return "";
  return str.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;").replace(/'/g, "&apos;");
}

async function generateVerseFrame(verse, outputPath, settings) {
  const {
    fontSize = 50,
    fontWeight = 700,
    fontFamily = "Amiri",
    textColor = "#ffffff",
    textPosition = "center",
    surahName = "",
  } = settings;

  // حسابات ديناميكية للخط - متطابقة مع الموقع
  // في الموقع: fontSize * 1.8 للفيديو 720px
  // في السيرفر: نفس المقياس
  const scaledFontSize = Math.min(fontSize * 1.6, 130); // حد أقصى لمنع الخروج عن الإطار
  const translationFontSize = 36;
  const lineSpacing = scaledFontSize * 1.4;
  const tLineSpacing = translationFontSize * 1.5;
  // 85% من عرض الفيديو للنص — 7.5% هامش من كل جانب
  const textAreaWidth = Math.floor(WIDTH * 0.85); // ~612px من 720px

  // تقسيم النص
  const verseLines = wrapArabicText(verse.text || "", scaledFontSize, textAreaWidth);
  const translationLines = wrapArabicText(verse.translation || "", translationFontSize, textAreaWidth);

  // حساب الارتفاع الكلي للمحتوى
  const verseBlockHeight = verseLines.length * lineSpacing;
  const badgeHeight = 50;
  const separatorSpace = 40;
  const translationBlockHeight = translationLines.length * tLineSpacing;
  const ornamentSpace = 70;
  
  const totalContentHeight = badgeHeight + 30 + verseBlockHeight + separatorSpace + translationBlockHeight + ornamentSpace;

  // حساب Y البداية بناءً على textPosition
  let contentStartY;
  if (textPosition === "top") {
    contentStartY = 120;
  } else if (textPosition === "bottom") {
    contentStartY = HEIGHT - totalContentHeight - 120;
  } else {
    contentStartY = (HEIGHT - totalContentHeight) / 2;
  }

  // بناء خطوط الآية SVG
  const verseStartY = contentStartY + badgeHeight + 40;
  const verseTSpans = verseLines.map((line, i) => 
    `<tspan x="${WIDTH / 2}" dy="${i === 0 ? 0 : lineSpacing}">${escapeXml(line)}</tspan>`
  ).join("");

  // خط الفاصل
  const separatorY = verseStartY + verseBlockHeight + 20;

  // الترجمة
  const translationStartY = separatorY + separatorSpace;
  const translationTSpans = translationLines.map((line, i) => 
    `<tspan x="${WIDTH / 2}" dy="${i === 0 ? 0 : tLineSpacing}">${escapeXml(line)}</tspan>`
  ).join("");

  // رقم الآية
  const ornamentY = translationStartY + translationBlockHeight + ornamentSpace;

  // CSS weight mapping
  const svgWeight = fontWeight >= 700 ? "bold" : fontWeight >= 500 ? "600" : "normal";
  
  const svg = `<svg width="${WIDTH}" height="${HEIGHT}" viewBox="0 0 ${WIDTH} ${HEIGHT}" xmlns="http://www.w3.org/2000/svg">
  <defs>
    <filter id="textShadow" x="-30%" y="-30%" width="160%" height="160%">
      <feDropShadow dx="0" dy="8" stdDeviation="15" flood-color="rgba(0,0,0,0.8)" flood-opacity="0.8"/>
    </filter>
    <filter id="softShadow" x="-20%" y="-20%" width="140%" height="140%">
      <feDropShadow dx="0" dy="3" stdDeviation="5" flood-color="rgba(0,0,0,0.6)" flood-opacity="0.6"/>
    </filter>
    <linearGradient id="goldGrad" x1="0%" y1="0%" x2="100%" y2="0%">
      <stop offset="0%" stop-color="#BF953F"/>
      <stop offset="25%" stop-color="#FCF6BA"/>
      <stop offset="50%" stop-color="#D4AF37"/>
      <stop offset="75%" stop-color="#FCF6BA"/>
      <stop offset="100%" stop-color="#AA771C"/>
    </linearGradient>
    <linearGradient id="darkGradient" x1="0" y1="0" x2="0" y2="1">
      <stop offset="0%" stop-color="rgba(0,0,0,0.4)"/>
      <stop offset="30%" stop-color="rgba(0,0,0,0.15)"/>
      <stop offset="70%" stop-color="rgba(0,0,0,0.2)"/>
      <stop offset="100%" stop-color="rgba(0,0,0,0.95)"/>
    </linearGradient>
  </defs>

  <!-- طبقة التعتيم المتدرج — مطابقة للموقع -->
  <rect width="${WIDTH}" height="${HEIGHT}" fill="url(#darkGradient)"/>

  <!-- بادج اسم السورة والنطاق -->
  ${surahName ? `
    <rect x="${(WIDTH - 220) / 2}" y="${contentStartY}" width="220" height="32" rx="16" 
          fill="rgba(255,255,255,0.05)" stroke="rgba(255,255,255,0.1)" stroke-width="1"/>
    <text x="${WIDTH / 2}" y="${contentStartY + 22}" 
          font-family="'${escapeXml(fontFamily)}', 'Amiri', serif" font-size="14" font-weight="bold" 
          fill="rgba(255,255,255,0.7)" text-anchor="middle" letter-spacing="1">
      ${escapeXml(surahName)} · ${escapeXml(String(verse.id))}
    </text>
  ` : ""}

  <!-- النص القرآني -->
  <text x="${WIDTH / 2}" y="${verseStartY}" 
        font-family="'${escapeXml(fontFamily)}', 'Amiri', serif" 
        font-size="${scaledFontSize}px" 
        font-weight="${svgWeight}" 
        fill="${escapeXml(textColor)}" 
        text-anchor="middle" 
        direction="rtl"
        filter="url(#textShadow)">
    ${verseTSpans}
  </text>

  <!-- خط فاصل ذهبي -->
  <rect x="${(WIDTH - 80) / 2}" y="${separatorY}" width="80" height="2" rx="1" fill="rgba(212,175,55,0.4)"/>

  <!-- الترجمة -->
  ${verse.translation ? `
    <text x="${WIDTH / 2}" y="${translationStartY}" 
          font-family="'${escapeXml(fontFamily)}', 'Amiri', serif" 
          font-size="${translationFontSize}px" 
          font-weight="500"
          fill="rgba(255,255,255,0.85)" 
          text-anchor="middle"
          font-style="italic"
          filter="url(#softShadow)">
      ${translationTSpans}
    </text>
  ` : ""}

  <!-- رقم الآية بالزخرفة -->
  <text x="${WIDTH / 2}" y="${Math.min(ornamentY, HEIGHT - 80)}" 
        font-family="'${escapeXml(fontFamily)}', 'Amiri', serif" 
        font-size="38px" font-weight="bold" 
        fill="url(#goldGrad)" text-anchor="middle">
    ﴿ ${verse.id} ﴾
  </text>
</svg>`;

  await sharp(Buffer.from(svg)).png({ quality: 100 }).toFile(outputPath);
}

// ═══════════════════════════════════════
// 🎬 محرك الرندرة الرئيسي — v4.0
// ═══════════════════════════════════════
async function renderCinematic(jobId, data) {
  const { verses, backgroundUrl, surahName, textColor, fontSize, fontWeight, fontFamily, textPosition } = data;
  const tempDir = path.resolve(os.tmpdir(), `work-${jobId}`);
  if (!fs.existsSync(tempDir)) fs.mkdirSync(tempDir, { recursive: true });
  const finalOutputPath = path.resolve(RENDERS_DIR, `${jobId}.mp4`);

  const settings = { fontSize, fontWeight, fontFamily, textColor, textPosition, surahName };

  try {
    jobs.set(jobId, { status: "processing", progress: 5, message: "💎 تحميل الموارد..." });

    // 1. تحميل الخط + الخلفية + الصوتيات بالتوازي
    const audioPaths = verses.map((_, i) => path.resolve(tempDir, `audio-${i}.mp3`));
    
    const [fontPath, bgPath] = await Promise.all([
      ensureFont(fontFamily, fontWeight),
      (async () => {
        if (!backgroundUrl) return null;
        const hash = crypto.createHash("md5").update(backgroundUrl).digest("hex").slice(0, 8);
        const isVid = /\.(mp4|webm|mov|m4v)/i.test(backgroundUrl);
        const ext = isVid ? "mp4" : "jpg";
        const dest = path.resolve(os.tmpdir(), `bg-${hash}.${ext}`);
        if (!fs.existsSync(dest)) await downloadFile(backgroundUrl, dest);
        return dest;
      })(),
      ...verses.map((v, i) => downloadFile(v.audio, audioPaths[i]))
    ]);

    jobs.set(jobId, { status: "processing", progress: 20, message: "🎨 تحليل الصوتيات..." });

    // 2. الحصول على مدة كل ملف صوتي
    const durations = await Promise.all(audioPaths.map(getAudioDuration));
    const totalDuration = durations.reduce((a, b) => a + b, 0);

    jobs.set(jobId, { status: "processing", progress: 30, message: "🖌️ رسم إطارات الآيات..." });

    // 3. توليد صورة PNG لكل آية
    const frameImages = [];
    for (let i = 0; i < verses.length; i++) {
      const imgPath = path.resolve(tempDir, `frame-${String(i).padStart(4, "0")}.png`);
      await generateVerseFrame(verses[i], imgPath, settings);
      frameImages.push({ path: imgPath, duration: durations[i] });
      
      const pct = 30 + Math.round((i / verses.length) * 20);
      jobs.set(jobId, { status: "processing", progress: pct, message: `🖌️ رسم آية ${i + 1} من ${verses.length}` });
    }

    jobs.set(jobId, { status: "processing", progress: 55, message: "🔊 دمج الصوت بدون تقطيع..." });

    // 4. ═══ دمج الصوت بدون أي فجوات ═══
    // استخدام FFmpeg filter_complex concat بدلاً من -f concat
    // هذا يمنع أي فراغات أو تقطيعات بين الآيات
    const mergedAudioPath = path.resolve(tempDir, "merged-audio.aac");
    
    // بناء أمر concat filter
    const audioInputs = audioPaths.map((p, i) => `-i "${p.replace(/\\/g, "/")}"`).join(" ");
    const filterParts = audioPaths.map((_, i) => `[${i}:a]aresample=44100,aformat=sample_fmts=fltp:channel_layouts=stereo[a${i}]`).join(";");
    const concatInputs = audioPaths.map((_, i) => `[a${i}]`).join("");
    const concatFilter = `${filterParts};${concatInputs}concat=n=${audioPaths.length}:v=0:a=1[aout]`;
    
    await execAsync(
      `ffmpeg ${audioInputs} -filter_complex "${concatFilter}" -map "[aout]" -c:a aac -b:a 192k -ar 44100 "${mergedAudioPath}" -y`,
      { maxBuffer: 50 * 1024 * 1024 }
    );

    jobs.set(jobId, { status: "processing", progress: 65, message: "🎬 تجميع الفيديو..." });

    // 5. ═══ بناء فيديو الإطارات ═══
    // ملف concat للصور مع المدد الدقيقة
    const frameListPath = path.resolve(tempDir, "frames.txt");
    let frameListContent = "";
    for (let i = 0; i < frameImages.length; i++) {
      const f = frameImages[i];
      frameListContent += `file '${f.path.replace(/\\/g, "/")}'\n`;
      frameListContent += `duration ${f.duration.toFixed(6)}\n`;
    }
    // آخر إطار مكرر (مطلوب من FFmpeg)
    frameListContent += `file '${frameImages[frameImages.length - 1].path.replace(/\\/g, "/")}'\n`;
    fs.writeFileSync(frameListPath, frameListContent);

    // 6. ═══ التجميع النهائي ═══
    const isVideoBg = bgPath && /\.(mp4|webm|mov|m4v)$/i.test(bgPath);
    
    let ffmpegCmd;
    
    if (bgPath) {
      // مع خلفية (صورة أو فيديو)
      const bgInputArgs = isVideoBg 
        ? `-stream_loop -1 -t ${totalDuration.toFixed(4)} -i "${bgPath.replace(/\\/g, "/")}"` 
        : `-loop 1 -t ${totalDuration.toFixed(4)} -i "${bgPath.replace(/\\/g, "/")}"`;
      
      ffmpegCmd = `ffmpeg ${bgInputArgs} -f concat -safe 0 -i "${frameListPath.replace(/\\/g, "/")}" -i "${mergedAudioPath.replace(/\\/g, "/")}" -filter_complex "[0:v]scale=${WIDTH}:${HEIGHT}:force_original_aspect_ratio=increase,crop=${WIDTH}:${HEIGHT},setsar=1[bg];[1:v]scale=${WIDTH}:${HEIGHT}:force_original_aspect_ratio=disable[fg];[bg][fg]overlay=0:0:format=auto,format=yuv420p[vout]" -map "[vout]" -map 2:a -c:v libx264 -preset ultrafast -crf 23 -c:a copy -t ${totalDuration.toFixed(4)} -movflags +faststart -y "${finalOutputPath.replace(/\\/g, "/")}"`;
    } else {
      // بدون خلفية (أسود)
      ffmpegCmd = `ffmpeg -f concat -safe 0 -i "${frameListPath.replace(/\\/g, "/")}" -i "${mergedAudioPath.replace(/\\/g, "/")}" -c:v libx264 -preset ultrafast -crf 23 -pix_fmt yuv420p -c:a copy -t ${totalDuration.toFixed(4)} -movflags +faststart -y "${finalOutputPath.replace(/\\/g, "/")}"`;
    }

    // تشغيل FFmpeg — استخدام execAsync للاستقرار على HF Spaces
    console.log(`🎬 [${jobId}] Running final FFmpeg merge (${totalDuration.toFixed(1)}s video)...`);
    jobs.set(jobId, { status: "processing", progress: 70, message: "🎬 جاري التصدير النهائي..." });
    
    await execAsync(ffmpegCmd, { maxBuffer: 100 * 1024 * 1024 });

    // ✅ نجاح!
    const host = "yousef891238-render-server.hf.space";
    jobs.set(jobId, { 
      status: "completed", 
      progress: 100, 
      url: `https://${host}/download/${jobId}.mp4`, 
      message: "✅ تم تصدير الفيديو بجودة سينمائية!" 
    });
    
    console.log(`✨ [${jobId}] Render completed successfully!`);

    // تنظيف بعد 60 ثانية
    setTimeout(() => {
      try { fs.rmSync(tempDir, { recursive: true, force: true }); } catch(e) {}
    }, 60000);

  } catch (err) {
    console.error(`❌ [${jobId}] Render failed:`, err);
    jobs.set(jobId, { status: "failed", error: err.message || "Unknown error" });
    // تنظيف فوري عند الفشل
    setTimeout(() => {
      try { fs.rmSync(tempDir, { recursive: true, force: true }); } catch(e) {}
    }, 5000);
  }
}
